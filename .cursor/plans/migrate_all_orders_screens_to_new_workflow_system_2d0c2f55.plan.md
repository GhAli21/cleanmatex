## Migration Strategy

### Phase 1: Preparation (Foundation)

#### 1.1 Create Screen Contract Helper Hook

**File:** `web-admin/lib/hooks/use-screen-orders.ts`

```typescript
import { useScreenContract } from './use-screen-contract';
import { useMemo } from 'react';

/**
                * Hook to get orders filtered by screen contract pre-conditions
 */
export function useScreenOrders(screen: string, options?: {
  page?: number;
  limit?: number;
  additionalFilters?: Record<string, any>;
  useOldWfCodeOrNew?: boolean;
}) {
  const { data: contract, isLoading: contractLoading } = useScreenContract(screen);
  
  const statusFilter = useMemo(() => {
    if (!contract?.preConditions?.statuses) return '';
    return contract.preConditions.statuses.join(',');
  }, [contract]);
  
  // Use existing orders API with screen contract statuses
  // This maintains backward compatibility
  return {
    statusFilter,
    contract,
    contractLoading,
  };
}
```

### Phase 3: Environment Variable Configuration

#### 3.1 Create Workflow System Config Utility

Create a centralized utility function to get the workflow system mode. This allows easy migration to feature flags or settings in the future by changing only one place.**File:** `web-admin/lib/config/workflow-config.ts`

```typescript
/**
    * Workflow System Configuration
    * 
    * Centralized function to determine which workflow system to use.
    * Currently reads from environment variable, but can be easily changed to:
    * - Feature flags
    * - Settings API
    * - Tenant-specific configuration
    * 
    * Change this function only to switch the source of the configuration.
 */

/**
    * Get the workflow system mode
    * @returns true to use new workflow system, false to use old workflow system
 */
export function getWorkflowSystemMode(): boolean {
  // Read from environment variable
  const envValue = process.env.NEXT_PUBLIC_USE_NEW_WORKFLOW_SYSTEM;
  
  // Default to false (use old system) for safety
  if (!envValue) return false;
  
  // Support both string and boolean values
  if (envValue === 'true' || envValue === '1') return true;
  if (envValue === 'false' || envValue === '0') return false;
  
  // Default to false if value is unclear
  return false;
}

/**
    * Hook version for React components (client-side)
    * Note: This reads from window.env or process.env.NEXT_PUBLIC_*
 */
export function useWorkflowSystemMode(): boolean {
  if (typeof window === 'undefined') {
    // Server-side: read from process.env
    return getWorkflowSystemMode();
  }
  
  // Client-side: read from NEXT_PUBLIC_ env var (available at build time)
  const envValue = process.env.NEXT_PUBLIC_USE_NEW_WORKFLOW_SYSTEM;
  
  if (!envValue) return false;
  if (envValue === 'true' || envValue === '1') return true;
  if (envValue === 'false' || envValue === '0') return false;
  
  return false;
}
```

**Environment Variable Setup:**Add to `.env.local`:

```bash
# Workflow System Configuration
# Set to 'true' to use new workflow system, 'false' to use old workflow system
# Default: false (uses old system)
NEXT_PUBLIC_USE_NEW_WORKFLOW_SYSTEM=false
```

**Future Migration Path:**To change to feature flags or settings, only modify `getWorkflowSystemMode()`:

```typescript
// Example: Change to feature flags
export function getWorkflowSystemMode(): boolean {
  // Call feature flags API
  const flags = getFeatureFlags();
      return flags.USE_NEW_WORKFLOW_SYSTEM || false;
}

// Example: Change to settings
export function getWorkflowSystemMode(): boolean {
  // Call settings API
  const settings = getEffectiveSettings();
  return settings.workflow.use_new_system || false;
}
```



#### 3.2 Update Screens to Use Config Utility

Each screen should use the config utility for `useOldWfCodeOrNew`:

```typescript
import { useWorkflowSystemMode } from '@/lib/config/workflow-config';
import { useOrderTransition } from '@/lib/hooks/use-order-transition';

export default function PreparationPage() {
  const useNewWorkflow = useWorkflowSystemMode();
const transition = useOrderTransition();

const handleTransition = async (orderId: string) => {
  await transition.mutateAsync({
    orderId,
    screen: 'preparation',
    input: {},
      // IMPORTANT (repo semantics):
      // - `useOldWfCodeOrNew === false` => OLD workflow
      // - `useOldWfCodeOrNew !== false` => NEW (enhanced) workflow
      //
      // So we pass the "use new workflow" boolean directly:
      useOldWfCodeOrNew: useNewWorkflow,
  });
};
}
```

**Note:** Despite the name `useOldWfCodeOrNew`, the actual code path selection is implemented as:

- `false` => OLD
- `true` (or undefined) => NEW

So the safest gradual rollout is to default the config to `false` and explicitly set `useOldWfCodeOrNew: false` until each screen is verified.
